    module priencr#(
    parameter WIDTH = 64
)(
    input   [WIDTH-1        : 0]    decode,
    output  [$clog2(WIDTH)-1 : 0]    encode,
    output                          valid
);

    generate
        if(WIDTH == 2'd2) begin
            assign  valid   = |decode;
            assign  encode  = decode[1];
        end
        else if(WIDTH & (WIDTH -1)) begin   // WIDTH is not a power of two, pad out the input WIDTH zero bits to the next power of two WIDTH
            priencr #(1<<$clog2(WIDTH)) priencer ({1<<$clog2(WIDTH){1'b0}} | decode, encode, valid);
        end
        else begin
            wire [$clog2(WIDTH)-2 : 0] encode_low;
            wire [$clog2(WIDTH)-2 : 0] encode_high;

            wire valid_low, valid_high;

            priencr #(WIDTH >> 1) priencr_low(decode[(WIDTH>>1)-1 : 0], encode_low, valid_low);
            priencr #(WIDTH >> 1) priencr_high(decode[WIDTH-1 : WIDTH>>1], encode_high, valid_high);

            assign valid    =   valid_low | valid_high;
            assign encode   =   valid_high ? {1'b1,encode_high} : {1'b0,encode_low};
        end
    endgenerate

endmodule
