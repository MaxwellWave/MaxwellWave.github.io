    查找首1的三种方法

    IC verilog
    Created At : 2023-11-13 10:00

    0 需求

    使用verilog实现搜索从高位到低位的数据中第一个1’b1位，并输出所对应的位号码

    1 分析

    思路1：如果已知数据的位宽，且数据位宽较小，则可以直接采用case语句进行查找实现

    思路2：对于一串查找首1的输入数据，如果已知其固定位宽，那么可以通过二分法的方式进行查找，但是二分法又必须保证是2的幂次方才可以循环判断。如8可以拆成8，4，2三级，但如果是10的话，二分法拆到第二级5就无法继续往下拆。此时可以通过高位补0的方式继续进行二分法拆分方式。根据此理论，可以通过实现一个递归判断的模块进行查找。即需要判断当前数据位是奇数还是偶数，对奇数进行补零操作后再送入递归模块，直至划分结束。

    思路3：可建立一个寄存器数组，通过generate实现多个组合逻辑对输入数据进行bit轮询查找，从LSB到MSB，并将每个bit的轮询结果存储在数组中，而最终的结果就是数组的最后一个

    2实现

    2.1 case查找法

    查找最高位的1'b1采用case语查找的话实际上就相当于一个优先编码器的实现，读者可以思考一下83优先编码器是如何实现的？咱们这个需求是不是和83优先编码器类似，故而我们可以采用如下的代码进行实现。其输入数据r_i_data为32bit，输出首1的位置int

    always @ (posedge clk) begin
 casez(r_i_data) 
        {{00{1'b0}},1'b1,{(I_WIDTH-01){1'b?}}}: begin int_d <= I_WIDTH - 01; end
        {{01{1'b0}},1'b1,{(I_WIDTH-02){1'b?}}}: begin int_d <= I_WIDTH - 02; end 
        {{02{1'b0}},1'b1,{(I_WIDTH-03){1'b?}}}: begin int_d <= I_WIDTH - 03; end 
        {{03{1'b0}},1'b1,{(I_WIDTH-04){1'b?}}}: begin int_d <= I_WIDTH - 04; end 
        {{04{1'b0}},1'b1,{(I_WIDTH-05){1'b?}}}: begin int_d <= I_WIDTH - 05; end 
        {{05{1'b0}},1'b1,{(I_WIDTH-06){1'b?}}}: begin int_d <= I_WIDTH - 06; end 
        {{06{1'b0}},1'b1,{(I_WIDTH-07){1'b?}}}: begin int_d <= I_WIDTH - 07; end 
        {{07{1'b0}},1'b1,{(I_WIDTH-08){1'b?}}}: begin int_d <= I_WIDTH - 08; end 
        {{08{1'b0}},1'b1,{(I_WIDTH-09){1'b?}}}: begin int_d <= I_WIDTH - 09; end 
        {{09{1'b0}},1'b1,{(I_WIDTH-10){1'b?}}}: begin int_d <= I_WIDTH - 10; end 
        {{10{1'b0}},1'b1,{(I_WIDTH-11){1'b?}}}: begin int_d <= I_WIDTH - 11; end 
        {{11{1'b0}},1'b1,{(I_WIDTH-12){1'b?}}}: begin int_d <= I_WIDTH - 12; end 
        {{12{1'b0}},1'b1,{(I_WIDTH-13){1'b?}}}: begin int_d <= I_WIDTH - 13; end 
        {{13{1'b0}},1'b1,{(I_WIDTH-14){1'b?}}}: begin int_d <= I_WIDTH - 14; end 
        {{14{1'b0}},1'b1,{(I_WIDTH-15){1'b?}}}: begin int_d <= I_WIDTH - 15; end 
        {{15{1'b0}},1'b1,{(I_WIDTH-16){1'b?}}}: begin int_d <= I_WIDTH - 16; end 
        {{16{1'b0}},1'b1,{(I_WIDTH-17){1'b?}}}: begin int_d <= I_WIDTH - 17; end 
        {{17{1'b0}},1'b1,{(I_WIDTH-18){1'b?}}}: begin int_d <= I_WIDTH - 18; end 
        {{18{1'b0}},1'b1,{(I_WIDTH-19){1'b?}}}: begin int_d <= I_WIDTH - 19; end 
        {{19{1'b0}},1'b1,{(I_WIDTH-20){1'b?}}}: begin int_d <= I_WIDTH - 20; end 
        {{20{1'b0}},1'b1,{(I_WIDTH-21){1'b?}}}: begin int_d <= I_WIDTH - 21; end 
        {{21{1'b0}},1'b1,{(I_WIDTH-22){1'b?}}}: begin int_d <= I_WIDTH - 22; end 
        {{22{1'b0}},1'b1,{(I_WIDTH-23){1'b?}}}: begin int_d <= I_WIDTH - 23; end 
        {{23{1'b0}},1'b1,{(I_WIDTH-24){1'b?}}}: begin int_d <= I_WIDTH - 24; end 
        {{24{1'b0}},1'b1,{(I_WIDTH-25){1'b?}}}: begin int_d <= I_WIDTH - 25; end 
        {{25{1'b0}},1'b1,{(I_WIDTH-26){1'b?}}}: begin int_d <= I_WIDTH - 26; end 
        {{26{1'b0}},1'b1,{(I_WIDTH-27){1'b?}}}: begin int_d <= I_WIDTH - 27; end 
        {{27{1'b0}},1'b1,{(I_WIDTH-28){1'b?}}}: begin int_d <= I_WIDTH - 28; end 
        {{28{1'b0}},1'b1,{(I_WIDTH-29){1'b?}}}: begin int_d <= I_WIDTH - 29; end 
        {{29{1'b0}},1'b1,{(I_WIDTH-30){1'b?}}}: begin int_d <= I_WIDTH - 30; end 
        {{30{1'b0}},1'b1,{(I_WIDTH-31){1'b?}}}: begin int_d <= I_WIDTH - 31; end 
        {{31{1'b0}},1'b1,{(I_WIDTH-32){1'b?}}}: begin int_d <= I_WIDTH - 32; end 
        default:begin int_d <= {$clog2(I_WIDTH){1'b0}}; end
 endcase
end

    在verilog中使用casez需要十分小心，这并非是说casez不会被综合，只是说综合工具可能无法识别z和？。上述代码笔者在vivado上完成了综合验证，但是大家还是可以思考一下，如果上述case语句实现的优先编码器采用if-else语句如何实现?如果使用反向case语句又该如何实现？大家有空可阅读一下《Verilog编程艺术》第14章 ,里面有我们想要的答案。

    2.2 二分法

    所谓的二分法即对输入数据依次对半划分，通过或语句来确定是否存在1'b1，从而确定当前划分位置的值，例如，如果输入8bit，那么分为前4bit和后4bit，如果前4bit中有1'b1存在，那么输出的位置至少大于4，即pos=3'b1xx，确认输出位置的最高位。然后再将前4bit划分为两个2bit，确认pos的第一位数据（从第零位计数），依次迭代向下划分。

    以输入8bit数据为例，如果需要确定其首1的位置，那么可通过如下代码进行实现

    module find_msb_one
(
    input   [7:0]   in_data,
    output  [2:0]   pos,
    output          vld
);
    wire   [3:0]    data_s1;
    wire   [1:0]    data_s2;
    
    //stage 1
    assign pos[2]  = |in_data[7:4] ? 1'b1 : 1'b0;
    assign data_s1 = pos[2] ? in_data[7:4] : in_data[3:0];

    //stage 2
    assign pos[1]  = |data_s1[2:1] ? 1'b1 : 1'b0;
    assign data_s2 = pos[1] ? data_s1[2:1] : data_s1[1:0];
    
    //stage 3
    assign pos[0]  = data_s2[1];

    // if in_data have not 1, the valid set low
    assign vld = |in_data;
endmodule

    以上代码只适合对于已知的输入数据且数据位宽为2的幂次方，如果数据不符合此条件又该如何进行处理呢？即能否设计一个可通过参数配置的模块实现此功能，便于模块的复用。

    在博客Recursive Modules | Beyond Circuits中博主提供了一种递归模块的设计思路，即在当前模块中例化本模块来进行实现递归运算。其设计框图及代码如下

    module priencr#(
    parameter WIDTH = 64
)(
    input   [WIDTH-1        : 0]    decode,
    output  [$clog2(WIDTH)-1 : 0]    encode,
    output                          valid
);

    generate
        if(WIDTH == 2'd2) begin
            assign  valid   = |decode;
            assign  encode  = decode[1];
        end
        else if(WIDTH & (WIDTH -1)) begin   // WIDTH is not a power of two, pad out the input WIDTH zero bits to the next power of two WIDTH
            priencr #(1<<$clog2(WIDTH)) priencer ({1<<$clog2(WIDTH){1'b0}} | decode, encode, valid);
        end
        else begin
            wire [$clog2(WIDTH)-2 : 0] encode_low;
            wire [$clog2(WIDTH)-2 : 0] encode_high;

            wire valid_low, valid_high;

            priencr #(WIDTH >> 1) priencr_low(decode[(WIDTH>>1)-1 : 0], encode_low, valid_low);
            priencr #(WIDTH >> 1) priencr_high(decode[WIDTH-1 : WIDTH>>1], encode_high, valid_high);

            assign valid    =   valid_low | valid_high;
            assign encode   =   valid_high ? {1'b1,encode_high} : {1'b0,encode_low};
        end
    endgenerate

endmodule

    如上述代码所示，整个priencr通过generate模块分为三个部分，首先优先级最高的是递归模块的结尾，即当输入数据位宽为2bit的时候，此时无需继续向下划分；其次是当划分的数据位宽是非2的幂次方的时候，通过WIDTH & (WIDTH -1)来判断，如果确实为非2的幂次方，则需要对其拓展为2的幂次方后再送入递归模块；最后则是将数据进行二分后送入递归模块。

    递归模块可以理解为从输出位置的LSB到MSB的定位，即其结果是从低位到高位确认，先确认最低位的值，然后送入上一个模块确认第二位的值，然后再送入上一个模块确认第三位的值，依次迭代。每一位的值由valid_high确认

    在上述priencr包含一些知识点需要掌握使用

    • if-generate的使用，特别是迭代例化，可以通过Verdi仿真深入理解其结构
    • $clog2是verilog-2001增加的一个系统函数，用于返回以2为底参数的对数的向上取整的值，即`clog2(x) = round_up(log2(x)) , 参数0返回值为0。一般用于参数的定义及传递
    • 判断一个数据是否为2的幂次方时，可采用WIDTH & (WIDTH-1)进行判断，因为如果是2的幂次方的话，则WIDTH的最高位为1，其余位为0;而WIDTH-1的最高位为0，其余位为1，此时两者相与为0
    • 对于迭代模块必须通过一个top模块进行例化调用，否则综合会出错

    2.3 bit 轮询法

    bit轮询法的设计思路很简单，就是从LSB到MSB依次对每个位置的数据进行判断是否为1，如果为1则将当前的位置保存进数组，否则不变，所有数据遍历完成后，数组的最后一个数据即为首1的位置，具体实现框图及代码如下(参考资料：verilog如何找一个n位数第m个1的位置? - 知乎)

    module highbit #(
    parameter OUT_WIDTH = 4, // out uses one extra bit for not-found
    parameter IN_WIDTH = 1<<(OUT_WIDTH-1)
) (
    input       [IN_WIDTH    -  1  :  0]                  in                    ,
    output      [OUT_WIDTH   -  1  :  0]                  out
);

    wire [OUT_WIDTH-1:0]out_stage[0:IN_WIDTH];
    assign out_stage[0] = {OUT_WIDTH{1'b1}}; // desired default output if no bits set
 generate genvar i;
     for(i=0; i<IN_WIDTH; i=i+1)
         assign out_stage[i+1] = in[i] ? i : out_stage[i]; 
 endgenerate
    assign out = out_stage[IN_WIDTH];

endmodule

    3 总结

    从上述三种实现方法来看，case查找法适用于已知小位宽的应用场景，但模块不可复用；二分法迭代适合各类情况，同样消耗资源也相对较多，并行级数只有log2(INPUT_WIDTH)；bit轮询法实现逻辑简单，适合各类情况，但是组合逻辑级数太长，级数有INPUT_WIDTH，影响电路的工作主频。

    4 思考

    如果需要找到n位数的第m个1的位置又该如何实现？显然此时二分法已不适用，case的优先编码器可无法考虑各类情况，verilog如何找一个n位数第m个1的位置? - 知乎中给出了一个方法，具体框图和代码如下，即将每次找到1的结果进行累加，然后再根据累加结果进行判断。

    module nth_one#(
 parameter IN_WIDTH = 4,
    parameter OUT_WIDTH = $clog2(IN_WIDTH)+1, // out uses one extra bit for not-found
    parameter N_TH_ONE = 3
   
) (
    input       [IN_WIDTH    -  1  :  0]                  in                    ,
    output      [OUT_WIDTH   -  1  :  0]                  out
);

    wire [OUT_WIDTH-1:0]out_stage[0:IN_WIDTH];
 
          assign out_stage[0] = {OUT_WIDTH{1'b1}}; // desired default output if not found
 generate genvar i;
     for(i=0; i<IN_WIDTH; i=i+1) begin:stage
  wire [i:0] ones; 
  wire            condition;
  assign     condition    =        (stage[i].ones== N_TH_ONE)&in[(IN_WIDTH -1 - i)];
  if(i == 0) begin
   assign stage[i].ones=         {OUT_WIDTH{1'b1}};  
  end else begin
   assign stage[i].ones=         stage[i-1].ones + in[(IN_WIDTH -1 - i)]; //increase 1 
  end
               assign out_stage[i+1]                 =                 condition? ((IN_WIDTH-1)-i) : out_stage[i]; 
      end
  endgenerate
   assign out = out_stage[IN_WIDTH];
endmodule
